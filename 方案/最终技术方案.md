## 🔐 认证方案

### 方案1：卡密登录（推荐，最简单）

#### 用户流程
```
1. 用户购买 → 获得卡密（AGENT2026-ABC123）
2. 访问网站 → 点击"解锁内容"
3. 输入卡密 → 调用腾讯云函数验证
4. 验证成功 → 返回Token → 存储到localStorage
5. 自动解锁全站内容
```

#### 技术实现

**前端代码（Login.vue）**
```vue
<template>
  <div class="login-modal">
    <h2>🔓 解锁全部内容</h2>
    <input 
      v-model="key" 
      placeholder="请输入卡密"
      @keyup.enter="login"
    />
    <button @click="login">解锁</button>
    <p v-if="error">{{ error }}</p>
  </div>
</template>

<script setup>
import { ref } from 'vue';

const key = ref('');
const error = ref('');

async function login() {
  try {
    const response = await fetch('https://你的云函数URL/verify-key', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        key: key.value,
        deviceId: getDeviceId()
      })
    });
    
    const data = await response.json();
    
    if (data.success) {
      localStorage.setItem('auth_token', data.token);
      localStorage.setItem('key', key.value);
      window.location.reload();
    } else {
      error.value = data.message;
    }
  } catch (e) {
    error.value = '验证失败，请重试';
  }
}

function getDeviceId() {
  // 生成设备指纹
  return btoa([
    navigator.userAgent,
    screen.width + 'x' + screen.height,
    navigator.language
  ].join('|'));
}
</script>
```

**后端代码（腾讯云函数）**
```javascript
// cloud-functions/verify-key/index.js
const cloud = require('wx-server-sdk');
cloud.init({ env: '你的环境ID' });
const db = cloud.database();

exports.main = async (event) => {
  const { key, deviceId } = event;
  
  try {
    // 1. 查询卡密
    const result = await db.collection('keys')
      .where({ key })
      .get();
    
    if (result.data.length === 0) {
      return { success: false, message: '卡密不存在' };
    }
    
    const keyData = result.data[0];
    
    // 2. 检查状态
    if (keyData.status === 'banned') {
      return { success: false, message: '卡密已被封禁' };
    }
    
    // 3. 检查设备数
    const devices = keyData.devices || [];
    if (!devices.includes(deviceId) && devices.length >= 3) {
      return { success: false, message: '设备数已达上限（最多3台）' };
    }
    
    // 4. 绑定设备
    if (!devices.includes(deviceId)) {
      devices.push(deviceId);
      await db.collection('keys')
        .doc(keyData._id)
        .update({
          devices,
          last_used: new Date()
        });
    }
    
    // 5. 生成Token
    const token = generateToken(key);
    
    // 6. 记录使用
    await db.collection('usage_logs').add({
      key,
      deviceId,
      timestamp: new Date(),
      ip: event.requestContext?.sourceIp
    });
    
    return { 
      success: true, 
      token,
      message: '验证成功'
    };
    
  } catch (error) {
    console.error(error);
    return { success: false, message: '服务器错误' };
  }
};

function generateToken(key) {
  // 简单的Token生成（生产环境建议用JWT）
  return Buffer.from(key + ':' + Date.now()).toString('base64');
}
```

## 🔧 管理脚本

### 生成卡密
```javascript
// scripts/generate-keys.js
const cloud = require('wx-server-sdk');
cloud.init({ env: '你的环境ID' });
const db = cloud.database();

async function generateKeys(count = 100) {
  const keys = [];
  
  for (let i = 0; i < count; i++) {
    const key = 'AGENT2026-' + 
      Math.random().toString(36).substring(2, 8).toUpperCase();
    
    await db.collection('keys').add({
      key,
      status: 'active',
      max_devices: 3,
      devices: [],
      created_at: new Date()
    });
    
    keys.push(key);
    console.log(`生成卡密 ${i + 1}/${count}: ${key}`);
  }
  
  // 保存到文件
  require('fs').writeFileSync(
    'keys.txt', 
    keys.join('\n')
  );
  
  console.log(`\n✅ 已生成 ${count} 个卡密，保存到 keys.txt`);
}

generateKeys(100);
```

### 封禁卡密
```javascript
// scripts/ban-key.js
async function banKey(key) {
  await db.collection('keys')
    .where({ key })
    .update({
      status: 'banned',
      banned_at: new Date()
    });
  
  console.log(`✅ 已封禁卡密: ${key}`);
}

banKey('AGENT2026-ABC123');
```

### 查看统计
```javascript
// scripts/stats.js
async function getStats() {
  const total = await db.collection('keys').count();
  const active = await db.collection('keys')
    .where({ status: 'active' })
    .count();
  const used = await db.collection('keys')
    .where({ 
      devices: db.command.neq([]) 
    })
    .count();
  
  console.log(`
📊 卡密统计
- 总数: ${total.total}
- 激活: ${active.total}
- 已使用: ${used.total}
- 未使用: ${active.total - used.total}
  `);
}

getStats();
```
